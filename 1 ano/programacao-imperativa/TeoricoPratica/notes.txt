CFLAGS = -Wall -Wextra
LIBS = -ln
OBJS = main.o hello_world.o


Makefile

$@ == target
$< == primeiro pré requisito
$^ == todos os requisitos

%.o: %.c:
  gcc $(CFLAGS) -c $^ -o $@

  
hello_world.o: hello_world.c
  gcc  -c hello_world -o hello_world.o

hello: main.c hello_world.o
  gcc $(LIBS) $(CFLAGS) -o hello main.c hello_world.o


clean:
  @-rm *.o hello // o - é para caso SE FALHAR

dep2:
  @echo

dep2:
  @echo

vars: dep1 dep2
  echo '$<' $<
  echo '$^' $^
  echo '$@' $@


make -l Makefile hello

se usar Makefile, basta make hello
